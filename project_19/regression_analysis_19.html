# --step_1 : select the target column(independent varible) in this case 'MonthlyIncome'
monthly_income = df_clean['MonthlyIncome']
cm('dependent variable(y)')
display(monthly_income[0 : 5])

# --step_2 : select the dependent variable
# for this find correlation between target columns and all other columns
correlations = df_clean.corr().MonthlyIncome.abs().sort_values(ascending = False)
cm('correlation between dependent variable and independent variable')
display(correlations[0 : 5])

# any column with correlation > 4.0 are taken. in this case 'JobLevel', 'TotalWorkingYears', 'YearsAtCompany', 'Age'
# creating new data frame with selected independent and dependent variables
cm('dataframe with selected variables')
df_selected = df_clean[['MonthlyIncome','JobLevel', 'TotalWorkingYears', 'YearsAtCompany', 'Age']]
display(df_selected[0 : 5])

# --step_3: check multi-colliniarity
correlations_selected = df_selected.corr()
cm('correlations between columns in selected data frame')
display(correlations_selected)

# vizualizing the correlation matrix
cm('vizualizing correlations with heatmap')
plt.figure(figsize=(7, 5))
sns.heatmap(correlations_selected, annot=True, fmt='g', cmap='YlGnBu')
plt.title('vizualization of correlation matrix')
plt.show()
# as these independent variables are correlated to each other only one of these to be taken
# as job level has maximum correlation it is gonna be taken

# --step_4:define variables and make variables in form of sklearn model format i.e y = [y1, y2, ...], x = [[x1, x2, x3], [x1, x2, x3], ...]
y = monthly_income
y = np.array(y)
x_1 = df_selected['JobLevel']
x_1 = np.array(x_1)
cm('the independent and dependent variable')    
display(y[0 :5])    
x = []
for i in range(len(x_1)):
    ar = []
    ar.append(df_selected['JobLevel'][i])
    x.append(ar)
x = np.array(x)    
display(x[0 : 5])    
    
# --step_5: vizualize the selected dependent and independent variables for an intuative sense
cm('the vizualization of relationship of independent and dependent variables')
plt.figure(figsize = (5, 3), dpi = 100)
plt.subplots_adjust(wspace = 0.3, hspace = 0.5)

plt.subplot(1, 1, 1)
color = 'black'
plt.scatter(x_1, y)
plt.title('MonthlyIncome vs JobLevel, (Fig-1)')
plt.xlabel( 'JobLevel', color = color)
plt.ylabel('MonthlyIncome')
plt.tick_params(axis = 'x', labelcolor = color)
plt.grid()
plt.show()

# --spet_6: apply the algorith
from sklearn import linear_model
model = linear_model.LinearRegression()
fit = model.fit(x, y)
cm("the model is trained. here's the model summery")
print(fit) 

# --step_7: get the estimated parameters of the model
# 1.model coefficients
print('\nmodel_coefficients: ', model.coef_) 

# 2.model intercept
print('\nmodel_intercept: ',model.intercept_) 

# 3.model predictions
predictions = model.predict(x)
print('\nmodel predictions: ', predictions[0 : 5])

# 4.root mean square error
from sklearn.metrics import mean_squared_error
rmse = mean_squared_error(y, predictions) ** 0.5
print('\nroot mean square error: ', rmse) 

# 5.r square
from sklearn.metrics import r2_score
r_sqr = r2_score(y, predictions)
print('\nr square:', r_sqr)
